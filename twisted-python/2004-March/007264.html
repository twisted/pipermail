     def __init__(self, tokens):
         self.waiting = []
         self.tokens  = tokens
	self.limit   = tokens

         """Attempt to acquire the token.

         @return Deferred which returns on token acquisition.
         """

         Should be called by whoever did the acquire() when the shared
