Hmm.  The following should work...


  dbpool = adbapi.ConnectionPool("MySQLdb",
                                host=host,
                                db='MYDB',
                                user='USER',
                                passwd='PASS')
  sql = """SELECT nom FROM dbo_questionnaire limit 0,50"""

  def consumer()
      query = flow.Generator(QueryIterator(dbpool, sql))
      while 1:
          yield query
          if query.stop: break
          print query.result

  flow.Flow(consumer)
  flow.execute()
