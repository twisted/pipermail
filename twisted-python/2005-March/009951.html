I do something similar with the help of a shell script - roughly like
 
#!/bin/sh
touch START_SERVER
while [ -f START_SERVER ]
do
  rm START_SERVER
  thisDir=`pwd | gawk -F '/' '{print $NF}'`
  twistd --pidfile=${thisDir}.pid -noy che.tac > /dev/null &
done
echo Server ${thisDir} stopped.
