straight out of
http://twistedmatrix.com/documents/current/howto/threading)

from twisted.internet import threads

def doLongCalculation():
  #time.sleep(3)
  return 3

def printResult(x):
  print "deferred returned "+str(x)

d.addCallback(printResult)
print "just waiting for deferred..."
