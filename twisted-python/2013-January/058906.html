Sigh. Make that:

    @inlineCallbacks
    def login(self, *args, **kwargs):
        for portal in self.portals:
            try:
                returnValue(yield portal.login(*args, **kwargs))
            except:
                continue
        # Ran out of portals, failed to login:
        raise Unauthorized()
