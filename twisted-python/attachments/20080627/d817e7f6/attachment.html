from twisted.internet import defer, reactor, task<br>
<br>
class _LoopUntil(object):<br>
<br>
    def __init__(self, delay, check):<br>
        self.delay = delay<br>
        self.check = check<br>
        self.result = defer.Deferred()<br>
<br>
    def go(self):<br>
        task.deferLater(reactor, self.delay, check).addCallbacks(<br>
            self._gotResult, self._failed)<br>
<br>
    def _gotResult(self, r):<br>
         if r:<br>
             d = self.result<br>
             del self.result<br>
             self.result.callback(r)<br>
         else:<br>
             self.go()<br>
<br>
    def _failed(self, f):<br>
        self.result.errback(f)<br>
        del self.result<br>
<br>
<br>
