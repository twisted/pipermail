class RemoteVOB:
     def __init__(self, vc, tag, site):
         self.vc = vc
         self.tag = tag
         self.site = site

     def make(self):
         return self.vc.callRemote("makeVOB", self.tag, self.site)
     def export(self, s):
         return self.vc.callRemote("exportVOB", self.tag, self.site, s)
     def import(self):
         return self.vc.callRemote("importVOB", self.tag, self.site)
