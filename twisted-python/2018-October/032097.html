This, however, DID work.

class tServer(Resource):
    isLeaf = False

    def getChild(self, path, request):
        print('You know what you doing.')

        if path == b'':
            print("Rendering /")
            return self

        return Resource.getChild(self, path, request)

    def render_GET(self, request):
        return bytes(content, "utf-8")
