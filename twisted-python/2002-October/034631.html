import pyPgSQL.PgSQL

      template render(self, request):
               print "DSN = %s" % (DSN,)

               ORDER BY release_date desc, news_item_id DESC
               LIMIT %s"""
               cursor.execute(sql, 8)

               <html>

                """
                </p>
                </body>
                </html>
                """ % vars()

resource = BlogResource()
