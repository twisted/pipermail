+        thing = thing._parents[0]

+        thing = thing.im_func

+        for x in vars(thing).itervalues():

+                newso = sourceOrder(x)
+            except TypeError:
+                # Not a sourceorderable
+                pass
+            else:
+                if so is not None:
+                    so = min(so, newso)
+                else:
+                    so = newso

+            return 0
+        return so
