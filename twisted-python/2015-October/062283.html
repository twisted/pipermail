memory profiling.

@profile
def run():
     d = {}
     l = []
     hp = hpy()
     before = hp.heap()            => *start* memory profiling from here

     d["k1"] = 'val1'
     d["k2"] = 10
     count = 0

         l.append(count)
         print 'The count is:', count
         count = count + 1
     print "Good bye!"
     after = hp.heap()         => *end* memory profiling here
     leftover = after - before

     print leftover
     del hp
